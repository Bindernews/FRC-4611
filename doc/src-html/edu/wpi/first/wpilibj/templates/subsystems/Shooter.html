001package edu.wpi.first.wpilibj.templates.subsystems;
002
003import edu.wpi.first.wpilibj.Victor;
004import edu.wpi.first.wpilibj.command.Subsystem;
005import edu.wpi.first.wpilibj.templates.RobotMap;
006import edu.wpi.first.wpilibj.templates.commands.ShooterMotorForward;
007import edu.wpi.first.wpilibj.templates.commands.ShooterMotorReverse;
008import edu.wpi.first.wpilibj.templates.commands.ShooterMotorStop;
009
010/**
011 * <p>One of the simplest possible subsystems that does anything. It consists of
012 * a single motor, this motor then controls opening and closing the motor
013 * through dead reckoning. It has three capabilities:</p>
014 * <ul>
015 * <li>{@link Claw#open() Open}: Tells the motor to open the claw.</li>
016 * <li>{@link Claw#close() Close}: Tells the motor to close the claw.</li>
017 * <li>{@link Claw#doNothing() Do Nothing}: Tells the motor to do nothing with
018 *                                          the claw.</li>
019 * </ul>
020 * 
021 * <p>Recommended next step: {@link Wrist}</p>
022 * 
023 * @author Alex Henning
024 */
025public class Shooter extends Subsystem {
026    // The claws motor
027    Victor motor;
028    
029    // Initialize your subsystem here
030    /**
031     * Initialize the claw with the motor declared in {@link RobotMap}
032     */
033    public Shooter() {
034        super("Shooter");
035        motor = new Victor(RobotMap.ShooterMotor);
036    }
037    
038    /**
039     * Initialize the default command to be {@link ClawDoNothing}.
040     */
041    public void initDefaultCommand() {
042        setDefaultCommand(new  ShooterMotorStop());
043    }
044    
045    /**
046     * Implements the claw's open capability. Simply tells the motor to move full
047     * speed in one direction, it is the commands responsibility to tell it when
048     * to stop opening. If the command doesn't, it will keep opening until the
049     * command ends and the default command {@link ClawDoNothing} takes over.
050     */
051    public void TurnOnForward() 
052        {
053        motor.set(1);
054    }
055    
056    
057    /**
058     * Implements the claw's close capability. Simply tells the motor to move
059     * full speed in one direction, it is the commands responsibility to tell it
060     * when to stop closing. If the command doesn't, it will keep opening until
061     * the command ends and the default command {@link ClawDoNothing} takes over.
062     */
063    public void TurnOnBackward() {
064        motor.set(-1);
065    }
066    
067    /**
068     * Implements the claw's do nothing capability. Simply tells the motor to
069     * stop moving. After this is called the claw is neither opening or closing.
070     */
071    public void TurnOff() {
072        motor.set(0);
073    }
074}
