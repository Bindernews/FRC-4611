001package edu.wpi.first.wpilibj.templates.subsystems;
002
003import edu.wpi.first.wpilibj.AnalogChannel;
004import edu.wpi.first.wpilibj.RobotDrive;
005import edu.wpi.first.wpilibj.command.PIDSubsystem;
006import edu.wpi.first.wpilibj.templates.RobotMap;
007import edu.wpi.first.wpilibj.templates.commands.CommandBase;
008import edu.wpi.first.wpilibj.templates.commands.DriveWithJoysticks;
009
010/**
011 * <p>The drive train is PID subsystem, but unlike the {@link Wrist} and
012 * {@link Elevator}, it is not always running PID. Instead, it can be run in a
013 * manual tank drive or PID can be enabled and it will use a range finder to
014 * drive a fixed distance away from the target.</p>
015 * 
016 * <p>Recommended next step: {@link CommandBase}</p>
017 *
018 * @author Alex Henning
019 */
020public class LeftDriveTrain extends PIDSubsystem {
021    // The constants for the P, I and D portion of PID
022    private static final double Kp = 3;
023    private static final double Ki = .2;
024    private static final double Kd = 0.0;
025    
026    RobotDrive leftdrive; 
027    AnalogChannel rangefinder;
028
029
030    // Initialize your subsystem here
031    public LeftDriveTrain() {
032        super("DriveTrain", Kp, Ki, Kd);
033        leftdrive = new RobotDrive(RobotMap.leftMotor1, RobotMap.leftMotor2); 
034        
035        
036        rangefinder = new AnalogChannel(RobotMap.rangefinder);
037    }
038    
039    /**
040     * Set the default command to drive with joysticks.
041     */
042    public void initDefaultCommand() {
043        setDefaultCommand(new DriveWithJoysticks());
044    }
045
046    /**
047     * @return The value of the rangefinder used as the PID input device.
048     *         These values correspond to your PID setpoint, in this case the
049     *         value can be anywhere between 0v and 5v.
050     */
051    protected double returnPIDInput() {
052        return rangefinder.getVoltage();
053    }
054
055    /**
056     * @param output The value to set the output to as determined by the PID
057     *               algorithm. This gets called each time through the PID loop
058     *               to update the output to the motor.
059     */
060    protected void usePIDOutput(double output) {
061        lefttankDrive(output, output);
062        }
063     
064    
065    /**
066     * Implements the tank drive capability of the drivetrain.
067     * 
068     * @param left The speed for the left side of the drivetrain.
069     * @param right The speed for the right side of the drivetrain.
070     */
071    public void lefttankDrive(double motor1, double motor2) {
072        leftdrive.tankDrive(motor1, motor2);
073        } 
074}
