005/**
006 * The RobotMap is a mapping from the ports sensors and actuators are wired into
007 * to a variable name. This provides flexibility changing wiring, makes checking
008 * the wiring easier and significantly reduces the number of magic numbers
009 * floating around.
010 * 
011 * <p>One of the benefits of the RobotMap is you can write your code without 
012 * knowing the ports in advance and can easily check that everything is wired
013 * properly from one simple file. Additionally, there are less magic numbers
014 * floating around the code so it is easier to understand and debug.</p>
015 * 
016 * <p>Recommended next step: {@link Claw}</p>
017 * 
018 * @author Alex Henning
019 */
020public class RobotMap {
021    // Motors: This is the mapping of motors ports marked PWM on the DIO board
022    public static final int
023            leftMotor1 = 1,
024                        leftMotor2 = 2,
025            rightMotor1 = 3,
026                        rightMotor2 = 4,
027                        ShooterMotor = 5,
028            FeederMotor = 6,
029            VerticalAimMotor = 7;
030    
031    // Sensors: This is the mapping of the analog ports on the analog module
032    // to the relevant sensors.
033    public static final int
034            rangefinder = 8,
035            elevatorPot = 9,
036            wristPot = 10;
037    
038    // Additional mappings could be included, but are not used on this robot for
039    // the task of soda delivery.
040}
