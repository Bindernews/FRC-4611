001package edu.wpi.first.wpilibj.templates.commands;
002
003/**
004 * <p>Similar to SetWristSetpoint, but it also has to handle enabling and
005 * disabling the PID loop</p>
006 * 
007 * <p>Recommended next step: {@link PrepareToGrab}</p>
008 *
009 * @author Alex Henning
010 */
011public class DriveToDistance extends CommandBase {
012    double setpoint;
013    
014    /**
015     * Require the drive train and store the desired setpoint.
016     * 
017     * @param setpoint The desired setpoint for the drive train.
018     */
019    public DriveToDistance(double setpoint) {
020        requires(leftdrivetrain);
021        this.setpoint = setpoint;
022    }
023
024    // Called just before this Command runs the first time
025    /**
026     * Set the setpoint to the stored value and enable PID on the drivetrain.
027     */
028    protected void initialize() {
029        leftdrivetrain.setSetpoint(setpoint);
030        leftdrivetrain.enable();
031    }
032
033    // Called repeatedly when this Command is scheduled to run
034    protected void execute() {
035    }
036
037    // Make this return true when this Command no longer needs to run execute()
038    /**
039     * @return true when it's close enough to the setpoint
040         * Assumes Left drive keeps track of distance
041     */
042        
043    protected boolean isFinished() {
044                // put logic here to decide when distance has been met
045                // 
046        return Math.abs(leftdrivetrain.getPosition() - setpoint) < .02;
047    }
048
049    // Called once after isFinished returns true
050    /**
051     * When this command ends, disable the drivetrain's PID
052     */
053    protected void end() {
054        leftdrivetrain.disable();
055    }
056
057    // Called when another command which requires one or more of the same
058    // subsystems is scheduled to run
059    /**
060     * When this command exits, disable the drivetrain's PID
061     */
062    protected void interrupted() {
063        leftdrivetrain.disable();
064    }
065}
