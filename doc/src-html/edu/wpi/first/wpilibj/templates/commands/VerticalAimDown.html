001package edu.wpi.first.wpilibj.templates.commands;
002
003/**
004 * <p>Another command that requires the claw. However, this one runs for a set
005 * period of time and then it's finished and stops executing.</p>
006 * 
007 * <p>Recommended next step: {@link SetWristSetpoint}<br/>
008 * Related command: {@link CloseClaw}</p>
009 *
010 * @author Alex Henning
011 */
012public class VerticalAimDown extends CommandBase {
013    
014    /**
015     * Initialize the command so that it requires the claw and set the timeout
016     * for 0.8 seconds, limiting runtime.
017     */
018    public VerticalAimDown() {
019        requires(verticalaim);
020        this.setTimeout(.8);
021    }
022
023    // Called just before this Command runs the first time
024    protected void initialize() {
025    }
026
027    // Called repeatedly when this Command is scheduled to run
028    /**
029     * Tells the claw to open.
030     */
031    protected void execute() {
032        verticalaim.down();
033    }
034
035    // Make this return true when this Command no longer needs to run execute()
036    /**
037     * @return true when it times out, aka after four fifths of a second.
038     */
039    protected boolean isFinished() {
040        return this.isTimedOut();
041    }
042
043    // Called once after isFinished returns true
044    protected void end() {
045
046    }
047
048    // Called when another command which requires one or more of the same
049    // subsystems is scheduled to run
050    protected void interrupted() {
051    }
052}
