018/**
019 * <p>The CommandBase class is automatically generated when the project is created.
020 * However, whenever you create a new subsystem, you must create a public static
021 * reference to it in the CommandBase class. All commands (except for
022 * CommandGroups) should be subclasses of CommandBase.</p>
023 * 
024 * <p>Recommended next step: {@link ClawDoNothing}</p>
025 * 
026 * @author Alex Henning
027 */
028public abstract class CommandBase extends Command {
029    // CommandBase holds a static instance of OI
030    public static OI oi;
031    
032    // Instances of each subsystem
033         public static LeftDriveTrain leftdrivetrain = new LeftDriveTrain();
034         public static RightDriveTrain rightdrivetrain = new RightDriveTrain();
035         public static Shooter shooter = new Shooter();
036         public static VerticalAim verticalaim = new VerticalAim(); 
037         public static Feeder feeder = new Feeder(); 
038    
039    /**
040     * Call this command to properly finish initializing the CommandBase.
041     * This call is automatically included in the default template.
042     */
043    public static void init() {
044        oi = new OI();
045        
046        // Optional: Logs the currently running command for each subsystem in
047        //           the SmartDashboard. This can be useful for debugging.
048        SmartDashboard.putData(leftdrivetrain);
049        SmartDashboard.putData(rightdrivetrain);
050        SmartDashboard.putData(shooter);
051        SmartDashboard.putData(verticalaim);
052    }
053    
054    // Automatically created constructors.
055    public CommandBase(String name) { super(name); }
056    public CommandBase() { super(); }
057}
