001package edu.wpi.first.wpilibj.templates.commands;
002
003/**
004 * <p>The default command for the drive train. When no other command is running,
005 * the robot can be driven using the joysticks.</p>
006 * 
007 * <p>Recommended next step: {@link DriveToDistance}
008 *
009 * @author Alex Henning
010 */
011public class DriveWithJoysticks extends CommandBase {
012    /**
013     * Requires the drive train
014     */
015    public DriveWithJoysticks() {
016        requires(leftdrivetrain);
017                 requires(rightdrivetrain);
018    }
019
020    // Called just before this Command runs the first time
021    protected void initialize() {
022    }
023
024    // Called repeatedly when this Command is scheduled to run
025    /**
026     * Have the drivetrain drive tank drive with the latest values from joysticks.
027     */
028    protected void execute() { 
029        leftdrivetrain.lefttankDrive(oi.getLeftSpeed(), oi.getLeftSpeed());
030                rightdrivetrain.righttankDrive(oi.getRightSpeed(), oi.getRightSpeed());
031    }
032
033    // Make this return true when this Command no longer needs to run execute()
034    /**
035     * @return false, so that it executes forever or until another command
036     *         interrupts it.
037     */
038    protected boolean isFinished() {
039        return false;
040    }
041
042    // Called once after isFinished returns true
043    protected void end() {
044    }
045
046    // Called when another command which requires one or more of the same
047    // subsystems is scheduled to run
048    protected void interrupted() {
049    }
050}
